import { pgTable, text, timestamp, primaryKey } from "drizzle-orm/pg-core"

export const dealAssignees = pgTable(
  'deal_assignees',
  {
    dealId: text('deal_id').notNull().references(() => deals.id),
    userId: text('user_id').notNull().references(() => users.id),
