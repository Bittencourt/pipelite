**1. Create `src/lib/workflows/export-import.ts`:**

```typescript
import { z } from "zod"

// Secret detection: match header KEYS (not values) that typically hold secrets
const SECRET_HEADER_KEYS = /^(authorization|bearer|token|api[-_]?key|secret|x-api[-_]?key|x-token|password|x-secret)/i

export const workflowExportSchema = z.object({
  schemaVersion: z.literal("pipelite/v1"),
  name: z.string().min(1),
  description: z.string().nullable(),
  triggers: z.array(z.record(z.string(), z.unknown())),
  nodes: z.array(z.record(z.string(), z.unknown())),
})

export type WorkflowExport = z.infer<typeof workflowExportSchema>
```

Export function `stripSecrets(headers: Record<string, string>): Record<string, string>`:
- Iterate header entries
- If key matches `SECRET_HEADER_KEYS` regex, replace value with `"{{PLACEHOLDER}}"`
- Otherwise keep value as-is
- Return new object (do not mutate input)

Export function `serializeWorkflowForExport(workflow: { name: string; description: string | null; triggers: Record<string, unknown>[]; nodes: Record<string, unknown>[] }): WorkflowExport`:
- Deep-clone nodes array via `JSON.parse(JSON.stringify(nodes))`
- Walk each node: if node has `config.headers` (Record<string, string>), apply `stripSecrets` to it
- Return `{ schemaVersion: "pipelite/v1", name, description, triggers, nodes: cleanedNodes }`

Export function `validateWorkflowImport(data: unknown): { valid: true; data: WorkflowExport } | { valid: false; error: string }`:
- If data is not an object, return `{ valid: false, error: "Invalid file: The selected file is not valid JSON." }`
- Parse with `workflowExportSchema.safeParse(data)`
- If parse fails, return `{ valid: false, error: "Incompatible workflow: This file was not exported from Pipelite or uses an unsupported version." }`
- Validate that all node types are in the known set: `["action", "condition", "delay"]`
- Validate that action nodes have actionType in: `["http", "crm", "email", "notification", "transform", "webhook-response"]`
- If unknown types found, return `{ valid: false, error: "Incompatible workflow: This file contains unsupported node types." }`
- Return `{ valid: true, data: parsed.data }`

Export function `slugify(name: string): string`:
- Lowercase, replace non-alphanumeric with hyphens, collapse multiple hyphens, trim hyphens from ends
